Use formidable for simple file upload

  1. Create a package.json configuration file
You can use npm init -y to generate a package.json file.

  1. Installation dependence
    Use the npm install -S package name to update the package.json file while installing the dependent library.

  2. Here use express specific installation does not say, directly into the body

  1. var express = require("express");
  2. var router = express.Router();
  3. var fs = require("fs");
  4. var path = require("path");
  5. var formidable = require("formidable");
  6.  
  7.  / / Provide a routing interface When accessing the root directory of this route will take this processing function
  8. router.post("/", function(req, res, next) {
  9.  Console.log("Start file upload....");
  10.  // Use foridable form to parse the received request. You can write the file directly to the local 
  11. var form = new formidable.IncomingForm();
  12.  / / Set the editor
  13. form.encoding = "utf-8";
  14.  / / Set the file storage path must be written
  15. form.uploadDir = "./public/images/";
  16.  / / retain suffix
  17. form.keepExtensions = true;
  18.  / / Set a single file size limit
  19. form.maxFieldsSize = 5 * 1024 * 1024;
  20.  //form.maxFields = 1000; set so the sum of the size of the file
  21.  / / Use form to parse the request fields are all the fields outside the file files contain file information
  22. form.parse(req, function(err, fields, files) {
  23.   console.log(fields);
  24.      / / Error handling
  25.   if (err) {
  26.     res.json({
  27.       code: 0,
  28.       message: `${err.message}`
  29.     });
  30.     return;
  31.   }
  32.  
  33.      / / File is empty
  34.   if (!files | (Object.keys(files).length == 0)) {
  35.     res.json({
  36.       code: 0,
  37.              Message: `File cannot be empty.
  38.     });
  39.     return;
  40.   }
  41.  
  42.      / / Modify the file name
  43.   var dir = form.uploadDir;
  44.      / / Note that the file attribute of files is the parameter name of the user upload settings. If the fileName is set then you need to use files.FileName
  45.      // Print files will find an object file corresponds to a File {} This format I don't know very well Maybe it is a file. You can skip the File and use the files.file. property to get the property value.
  46.  
  47.   var oldName = files.file.path;
  48.   var newName = path.join(dir, files.file.name);
  49.      / / Modify the file name and suffix
  50.   fs.rename(oldName, newName, () => {
  51.     console.log(`renameed to ${newName}`);
  52.   });
  53.  
  54.   res.json({
  55.     code: 1
  56.   });
  57.  
  58. });
  59. });
  60.  
  61. module.exports = router;
  62.
